remarks:
remarks:        Where:
remarks:
remarks:        T = Relationship type
remarks:        1) Transit
remarks:        2) Peer: public
remarks:        3) Peer: private
remarks:        4) Peer: paid
remarks:        5) Customer: partial
remarks:        6) Customer: full
remarks:        7) Internal
remarks:
remarks:        RCC = Combination of region/country and city code
remarks:        where R is the region/country and CC is the city.
remarks:
remarks:        1xx) Europe (where xx is 00-99)
remarks:        101) London, UK
remarks:        102) Manchester, UK
remarks:        103) Amsterdam, NL
remarks:
remarks:        Example: To match all transit routes in Manchester
remarks:        use "^34270:11102$"
remarks:

remarks:        Europe use "^34270:1([2-4])1(.{2})$"
